def cross(colors, lista, lista2, disposizioni):
    if len(colors) == 2:
        for i in range(len(lista)):
            if i % 2:
                lista[i] = colors[1]
                lista2[i] = colors[0]
            else:
                lista[i] = colors[0]
                lista2[i] = colors[1]
        disposizioni.append(lista[:])
        disposizioni.append(lista2[:])
        return disposizioni
    else:
        for colore in colors[1:]:
            for i in range(len(lista)):
                if i % 2:
                    lista[i] = colore
                    lista2[i] = colors[0]
                else:
                    lista[i] = colors[0]
                    lista2[i] = colore
            disposizioni.append(lista[:])
            disposizioni.append(lista2[:])
        cross(colors[1:], lista, lista2, disposizioni)
        return disposizioni
